DATABASE SCHEMA

- USER MANAGEMENT

Users (
    UserId INT PK IDENTITY,
    FullName VARCHAR(100),
    Email VARCHAR(150) UNIQUE,
    Phone VARCHAR(32) UNIQUE,
    IsActive BIT DEFAULT 1,
    RoleId INT FK → Roles.RoleId,
    CreatedDate DATETIME DEFAULT GETDATE(),
    UpdatedDate DATETIME,
    PasswordHash VARCHAR(MAX)
)

Roles (
    RoleId INT PK IDENTITY,
    RoleName VARCHAR(20) UNIQUE  -- SuperAdmin, HR, Mentor, Panel
)

//UserRoles (
//    UserRoleId INT PK IDENTITY,
//    UserId INT FK → Users.UserId,
//    RoleId INT FK → Roles.RoleId
//)

UserPermissions (
    UserId INT FK → Users.UserId,
    Action VARCHAR(20) NOT NULL,
    View BIT NOT NULL,
    Add BIT NOT NULL,
    Update BIT NOT NULL,
    DELETE BIT NOT NULL
)

Requests (
    RequestId INT PK IDENTITY,
    RequestType VARCHAR(20)  -- Add Role, Add Action
    SubType VARCHAR(30) -- HR, Mentor, Panel, Drive-Add, Drive-View, Drive-Delete,
    Status VARCHAR(20) DEFAULT 'Pending' -- Pending, Approved, Rejected
    ApprovedBy INT NULL FK → Users.UserId,
    ApprovedDate DATETIME NULL,
    RequestedBy INT NOT NULL FK → Users.UserId,
    RequestedDate DATETIME DEFAULT GETDATE(),

)

- DRIVE MANAGEMENT

Drives (
    DriveId INT PK IDENTITY,
    DriveName VARCHAR(200) UNIQUE,
    DriveDate DATETIME,
    TechnicalRounds INT NOT NULL,
    CreatedBy INT NOT NULL FK → Users.UserId,
    Status VARCHAR(20) DEFAULT 'InProposal',   -- InProposal, Started, Halted, Completed
    CreatedDate DATETIME DEFAULT GETDATE()
)

DriveMembers (
    DriveMemberId INT PK IDENTITY,
    DriveId INT FK → Drives.DriveId,
    UserId INT FK → Users.UserId,
    RoleId INT FK → Roles.RoleId
)

- DRIVE CONFIGURATION

DriveRoleConfiguration (
    ConfigId INT PK IDENTITY,
    DriveId INT FK → Drives.DriveId,
    RoleId INT FK → Roles.RoleId,
    AllowPanelReassign BIT,
    CanViewFeedback BIT,
    AllowBulkUpload BIT,
    CanEditSubmittedFeedback BIT,
    RequireApprovalForReassignment BIT
)

PanelVisibilitySettings (
    VisibilityId INT PK IDENTITY,
    DriveId INT NOT NULL UNIQUE FK → Drives.DriveId,
    ShowPhone BIT,
    ShowEmail BIT,
    ShowPreviousCompany BIT,
    ShowResume BIT,
    ShowCollege BIT,
    ShowAddress BIT,
    ShowLinkedIn BIT,
    ShowGitHub BIT
)

NotificationSettings (
    NotificationId INT PK IDENTITY,
    DriveId INT NOT NULL UNIQUE FK → Drives.DriveId,
    EmailNotificationEnabled BIT
)

FeedbackConfiguration (
    FeedbackConfigId INT PK IDENTITY,
    DriveId INT NOT NULL UNIQUE FK → Drives.DriveId,
    OverallRatingRequired BIT,
    TechnicalSkillRequired BIT,
    CommunicationRequired BIT,
    ProblemSolvingRequired BIT,
    RecommendationRequired BIT,
    OverallFeedbackRequired BIT
)

- CANDIDATE MANAGEMENT

Candidates (
    CandidateId INT PK IDENTITY,
    FullName VARCHAR(100) NOT NULL,
    Email VARCHAR(150) NOT NULL UNIQUE,
    Phone VARCHAR(32) NOT NULL UNIQUE,
    Address VARCHAR(500) NULL,
    College VARCHAR(200) NULL,
    PreviousCompany VARCHAR(200) NULL,
    ExperienceLevel VARCHAR(20) NOT NULL,
    TechStack VARCHAR(MAX) NULL,
    ResumeUrl VARCHAR(500) NULL,
    LinkedInUrl VARCHAR(500) NULL,
    GitHubUrl VARCHAR(500) NULL,
    CreatedDate DATETIME DEFAULT GETDATE()
)

DriveCandidates (
    DriveCandidateId INT PK IDENTITY,
    CandidateId INT FK → Candidates.CandidateId,
    DriveId INT FK → Drives.DriveId,
    Status VARCHAR(20) DEFAULT 'Pending',  -- Pending, Selected, Rejected
    StatusSetBy INT NULL FK → Users.UserId,
    CreatedDate DATETIME DEFAULT GETDATE()
)

Rounds (
    RoundId INT PK IDENTITY,
    DriveCandidateId INT FK → DriveCandidates.DriveCandidateId,
    InterviewerId INT FK → Users.UserId,
    RoundType VARCHAR(20),  -- Hr, Tech 1, Tech 2
    Status VARCHAR(20) DEFAULT 'Scheduled',  -- Scheduled, OnProcess, Completed, Skipped
    Result VARCHAR(20) DEFAULT 'Pending',  -- Pending, Selected, Rejected
)

CandidateReassignments (
    ReassignId INT PK IDENTITY,
    DriveCandidateId INT NOT NULL FK → DriveCandidates.DriveCandidateId,
    PreviousUserId INT NOT NULL FK → Users.UserId,
    NewUserId INT NOT NULL FK → Users.UserId,
    RequestedBy INT NOT NULL FK → Users.UserId,
    RequireApproval BIT NOT NULL,
    ApprovedBy INT FK → Users.UserId NULL,
    ApprovedDate DATETIME NULL,
    RequestedDate DATETIME DEFAULT GETDATE()
)

- INTERVIEW & FEEDBACK

Interviews (
    InterviewId INT PK IDENTITY,
    DriveCandidateId INT NOT NULL FK → DriveCandidates.DriveCandidateId,
    InterviewerId INT NOT NULL FK → Users.UserId,
    InterviewDate DATETIME
)

Feedbacks (
    FeedbackId INT PK IDENTITY,
    InterviewId INT NOT NULL UNIQUE FK → Interviews.InterviewId,
    OverallRating INT NOT NULL,
    TechnicalSkill VARCHAR(500),
    Communication VARCHAR(500),
    ProblemSolving VARCHAR(500),
    OverallFeedback TEXT,
    Recommendation VARCHAR(20) NOT NULL, -- Hire, May be, No Hire
    SubmittedDate DATETIME DEFAULT GETDATE()
)
